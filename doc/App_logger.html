def initialize
  ensure_log_directory
  self.logger = Logger.new(LOG_FILE_PATH)
  self.logger.formatter = proc do |severity, datetime, progname, msg|
    "[#{datetime}] #{severity}: #{msg}\n"
  end
  setup_log_level
end
